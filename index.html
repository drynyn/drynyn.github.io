class Game { constructor() { this.level = 1; this.thoughts = 0; } upgrade() { this.thoughts -= this.level * 10; this.level += 1; } earnThoughts(thoughts) { this.thoughts += thoughts; } } const game = new Game(); const thoughtsElement = document.getElementById("thoughts"); const levelElement = document.getElementById("level"); function updateUI() { thoughtsElement.innerHTML = game.thoughts; levelElement.innerHTML = game.level; } function upgrade() { if(game.thoughts >= game.level * 10){ game.upgrade(); updateUI(); setInterval(() => { game.earnThoughts(game.level/10); updateUI(); }, 1000); } else { alert("Not Enough thoughts"); } } function meditate() { game.earnThoughts(1); updateUI(); }
